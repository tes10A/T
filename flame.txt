local function createSlider(labelText)
	local container = Instance.new("Frame")
	container.Size = UDim2.new(1, -30, 0, 50)
	container.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
	container.Parent = menu

	local cs = Instance.new("UIStroke", container)
	cs.Color = Color3.fromRGB(150, 150, 150)

	local cc = Instance.new("UICorner", container)
	cc.CornerRadius = UDim.new(0, 8)

	local sliderBar = Instance.new("Frame")
	sliderBar.Size = UDim2.new(0.6, 0, 0, 6)
	sliderBar.Position = UDim2.new(0.05, 0, 0.5, -3)
	sliderBar.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
	sliderBar.Parent = container

	local sliderCorner = Instance.new("UICorner", sliderBar)
	sliderCorner.CornerRadius = UDim.new(1, 0)

	local sliderKnob = Instance.new("Frame")
	sliderKnob.Size = UDim2.new(0, 14, 0, 14)
	sliderKnob.Position = UDim2.new(0, 0, 0.5, -7)
	sliderKnob.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
	sliderKnob.Parent = sliderBar

	local knobCorner = Instance.new("UICorner", sliderKnob)
	knobCorner.CornerRadius = UDim.new(1, 0)

	local inputBox = Instance.new("TextBox")
	inputBox.Size = UDim2.new(0.25, 0, 0.6, 0)
	inputBox.Position = UDim2.new(0.7, 0, 0.2, 0)
	inputBox.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
	inputBox.TextColor3 = Color3.new(1, 1, 1)
	inputBox.TextScaled = true
	inputBox.Text = "10"
	inputBox.Parent = container

	local boxCorner = Instance.new("UICorner", inputBox)
	boxCorner.CornerRadius = UDim.new(0, 4)

	local boxStroke = Instance.new("UIStroke", inputBox)
	boxStroke.Color = Color3.fromRGB(150, 150, 150)

	local value = 10

	local function updateSlider(newValue)
		value = math.max(10, math.min(100, newValue))
		local ratio = (value - 10) / 90
		sliderKnob.Position = UDim2.new(ratio, -7, 0.5, -7)
		inputBox.Text = tostring(value)
		return value
	end

	local draggingSlider = false

	sliderKnob.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			draggingSlider = true
		end
	end)

	sliderKnob.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			draggingSlider = false
		end
	end)

	UserInputService.InputChanged:Connect(function(input)
		if draggingSlider and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
			local barSize = sliderBar.AbsoluteSize.X
			local barPosition = sliderBar.AbsolutePosition.X
			local mousePos = input.Position.X
			local ratio = math.max(0, math.min(1, (mousePos - barPosition) / barSize))
			local newValue = 10 + (ratio * 90)
			updateSlider(newValue)
		end
	end)

	inputBox.FocusLost:Connect(function()
		local num = tonumber(inputBox.Text)
		if num then
			updateSlider(num)
		else
			inputBox.Text = tostring(value)
		end
	end)

	return container, function() return value end, function(newVal) updateSlider(newVal) end
end

local function toggleLogic(toggle, knob)
	local state = false
	toggle.MouseButton1Click:Connect(function()
		state = not state
		if state then
			knob:TweenPosition(UDim2.new(1, -22, 0.5, -10), "Out", "Quad", 0.15, true)
			toggle.BackgroundColor3 = Color3.fromRGB(0, 170, 0)
		else
			knob:TweenPosition(UDim2.new(0, 2, 0.5, -10), "Out", "Quad", 0.15, true)
			toggle.BackgroundColor3 = Color3.fromRGB(70, 70, 70)
		end
	end)
	return function() return state end
end

local function updateCrosshairSize()
	if crosshairGui then
		local size = getCrosshairSize() * 2
		local circle = crosshairGui:FindFirstChild("CrosshairCircle")
		if circle then
			circle.Size = UDim2.new(0, size, 0, size)
			circle.Position = UDim2.new(0.5, -size / 2, 0.5, -size / 2)
		end
	end
end

local function makePlayerInvisible()
	if player.Character then
		for _, part in ipairs(player.Character:GetDescendants()) do
			if part:IsA("BasePart") then
				part.Transparency = 1
			elseif part:IsA("Accessory") then
				local handle = part:FindFirstChild("Handle")
				if handle then
					handle.Transparency = 1
				end
			end
		end
	end
end

local function makePlayerVisible()
	if player.Character then
		for _, part in ipairs(player.Character:GetDescendants()) do
			if part:IsA("BasePart") and part.Parent:IsA("Humanoid") == false then
				if part.Parent.Name ~= "Accessory" and part.Parent:FindFirstChild("Humanoid") == nil then
					part.Transparency = 0
				end
			elseif part:IsA("Accessory") then
				local handle = part:FindFirstChild("Handle")
				if handle then
					handle.Transparency = 0
				end
			end
		end
		
		for _, part in ipairs(player.Character:GetChildren()) do
			if part:IsA("BasePart") then
				part.Transparency = 0
			elseif part:IsA("Accessory") then
				local handle = part:FindFirstChild("Handle")
				if handle then
					handle.Transparency = 0
				end
			end
		end
	end
end

local function isEnemy(targetPlayer)
	if targetPlayer.Character and player.Character then
		local targetHumanoid = targetPlayer.Character:FindFirstChild("Humanoid")
		local playerHumanoid = player.Character:FindFirstChild("Humanoid")
		
		if targetHumanoid and playerHumanoid then
			if targetHumanoid:FindFirstChild("Team") and playerHumanoid:FindFirstChild("Team") then
				return targetHumanoid.Team ~= playerHumanoid.Team
			else
				if targetPlayer:FindFirstChild("Team") and player:FindFirstChild("Team") then
					return targetPlayer.Team.Value ~= player.Team.Value
				end
			end
		end
	end
	return true
end

local function getEnemyInView()
	if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then
		return nil
	end

	local playerHrp = player.Character.HumanoidRootPart
	local cameraDirection = camera.CFrame.LookVector
	local closestEnemy = nil
	local closestAngle = 60

	for _, p in ipairs(Players:GetPlayers()) do
		if p ~= player and p.Character and isEnemy(p) then
			local enemyHrp = p.Character:FindFirstChild("HumanoidRootPart")
			if enemyHrp then
				local directionToEnemy = (enemyHrp.Position - playerHrp.Position).Unit
				local angle = math.acos(math.min(1, math.max(-1, cameraDirection:Dot(directionToEnemy)))) * (180 / math.pi)

				if angle < closestAngle then
					closestAngle = angle
					closestEnemy = p
				end
			end
		end
	end

	return closestEnemy
end

local function updateAuraHighlights()
	if not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") then
		return
	end

	local playerPos = player.Character.HumanoidRootPart.Position
	local maxDistance = 500

	for _, p in ipairs(Players:GetPlayers()) do
		if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
			local distance = (p.Character.HumanoidRootPart.Position - playerPos).Magnitude

			if distance <= maxDistance then
				if not auraHighlights[p] then
					local hl = Instance.new("Highlight")
					hl.FillColor = Color3.fromRGB(255, 0, 0)
					hl.OutlineColor = Color3.fromRGB(255, 0, 0)
					hl.Parent = p.Character
					auraHighlights[p] = hl
				end
			else
				if auraHighlights[p] then
					auraHighlights[p]:Destroy()
					auraHighlights[p] = nil
				end
			end
		end
	end
end

RunService.RenderStepped:Connect(function()
	if auraState() then
		local currentTime = tick()
		if currentTime - lastAuraUpdate >= auraUpdateInterval then
			lastAuraUpdate = currentTime
			updateAuraHighlights()
		end
	else
		for _, hl in pairs(auraHighlights) do
			if hl then hl:Destroy() end
		end
		auraHighlights = {}
	end

	if tpState() and selectedTpPlayer and selectedTpPlayer.Character and player.Character then
		local hrp = selectedTpPlayer.Character:FindFirstChild("HumanoidRootPart")
		local my = player.Character:FindFirstChild("HumanoidRootPart")
		if hrp and my then
			my.CFrame = hrp.CFrame
		end
	end

	if distState() and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
		for _, p in ipairs(Players:GetPlayers()) do
			if p ~= player and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
				local dist = (p.Character.HumanoidRootPart.Position - player.Character.HumanoidRootPart.Position).Magnitude
				if not distLabels[p] then
					local bill = Instance.new("BillboardGui")
					bill.Size = UDim2.new(0, 100, 0, 40)
					bill.AlwaysOnTop = true
					bill.Parent = p.Character

					local txt = Instance.new("TextLabel")
					txt.Size = UDim2.new(1, 0, 1, 0)
					txt.BackgroundTransparency = 1
					txt.TextColor3 = Color3.new(1, 0, 0)
					txt.TextScaled = true
					txt.Parent = bill

					distLabels[p] = txt
				end
				distLabels[p].Text = p.Name .. " Dist: " .. math.floor(dist)
				distLabels[p].Parent.Adornee = p.Character.HumanoidRootPart
			end
		end
	else
		for _, v in pairs(distLabels) do
			if v and v.Parent then
				v.Parent:Destroy()
			end
		end
		distLabels = {}
	end

	if crosshairState() then
		createCrosshair()
		updateCrosshairSize()
	else
		destroyCrosshair()
	end

	if aimbotState() then
		makePlayerInvisible()
		local targetEnemy = getEnemyInView()

		if targetEnemy and targetEnemy.Character and player.Character then
			aimbotTarget = targetEnemy
			local neckPos = targetEnemy.Character:FindFirstChild("Neck")
			local head = targetEnemy.Character:FindFirstChild("Head")
			local playerHrp = player.Character:FindFirstChild("HumanoidRootPart")
			
			if playerHrp then
				local aimPos = head.Position
				if neckPos then
					aimPos = neckPos.Position + (head.Position - neckPos.Position) * 0.3
				else
					aimPos = head.Position + Vector3.new(0, -3, 0)
				end
				camera.CFrame = CFrame.new(camera.CFrame.Position, aimPos)
			end
		end
	else
		aimbotTarget = nil
		makePlayerVisible()
		if player.Character and player.Character:FindFirstChild("Humanoid") then
			camera.CameraSubject = player.Character:FindFirstChild("Humanoid")
		end
	end
end)

Players.PlayerRemoving:Connect(function(leftPlayer)
	if auraHighlights[leftPlayer] then
		auraHighlights[leftPlayer]:Destroy()
		auraHighlights[leftPlayer] = nil
	end
	if distLabels[leftPlayer] then
		if distLabels[leftPlayer].Parent then
			distLabels[leftPlayer].Parent:Destroy()
		end
		distLabels[leftPlayer] = nil
	end
	if selectedTpPlayer == leftPlayer then
		selectedTpPlayer = nil
		boxTp.Text = ""
	end
end)

	wait(0.1)
	if not aimbotState() then
		makePlayerVisible()
	end
end)
